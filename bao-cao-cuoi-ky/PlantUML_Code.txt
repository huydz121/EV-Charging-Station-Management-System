  + managementPermissions: List
  --
  + manageCustomers()
  + manageStations()

  + stationId: String
  + name: String
  + address: String

  + updateStationStatus()

  + chargerId: String

  + chargerType: String
  + maxPower: Float
  + status: String

  + updateChargerStatus()

  + vehicleId: String
  + customerId: String

  + batteryCapacity: Float

  + maintenanceId: String
  + stationId: String
  + chargerId: String
  + maintenanceDate: Date
  + status: String

  + scheduleMaintenance()

  + reservationId: String
  + customerId: String
  + chargerId: String
  + reservedStartTime: DateTime
  + reservedEndTime: DateTime
  + status: String
  --
  + bookSlot()
  + cancelReservation()

  + incidentId: String
  + stationId: String
  + chargerId: String
  + description: String
  + timestamp: DateTime
  + status: String

  + updateStatus()

  + sessionId: String

  + chargerId: String

  + endTime: DateTime

  + totalPrice: Float
  + status: String

  + stopSession()

ChargingStation "1" -- "0..*" StationIncident : phát sinh >
Charger "1" -- "0..*" StationIncident : liên quan >

' Quan hệ Phụ thuộc
Administrator ..> Customer : quản lý >
Administrator ..> ChargingStation : quản lý >
